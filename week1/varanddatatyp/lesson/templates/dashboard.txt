    <script>
        let currentMetrics = {};
        
        // Update metrics every 2 seconds
        function updateMetrics() {
            fetch('/api/metrics')
                .then(response => response.json())
                .then(data => {
                    currentMetrics = data;
                    updateDashboard(data);
                })
                .catch(error => {
                    console.error('Error fetching metrics:', error);
                    showError('Failed to fetch metrics');
                });
        }
        
        function updateDashboard(data) {
            // Update status
            document.getElementById('agent-status').textContent = data.status;
            document.getElementById('agent-name').textContent = data.agent_name;
            document.getElementById('agent-version').textContent = data.version;
            
            // Update metrics
            document.getElementById('total-interactions').textContent = data.total_interactions;
            document.getElementById('success-rate').textContent = data.success_rate.toFixed(1) + '%';
            document.getElementById('avg-confidence').textContent = (data.average_confidence * 100).toFixed(1) + '%';
            document.getElementById('min-confidence').textContent = (data.min_confidence * 100).toFixed(1) + '%';
            document.getElementById('max-confidence').textContent = (data.max_confidence * 100).toFixed(1) + '%';
            
            // Update confidence bar
            const confidenceBar = document.getElementById('confidence-bar');
            const confidencePercent = data.average_confidence * 100;
            confidenceBar.style.width = confidencePercent + '%';
            
            // Update status indicator
            const statusIndicator = document.getElementById('status-indicator');
            if (data.is_active) {
                statusIndicator.className = 'status-indicator status-active';
            } else {
                statusIndicator.className = 'status-indicator status-inactive';
            }
        }
        
        function sendMessage() {
            const input = document.getElementById('user-input');
            const message = input.value.trim();
            
            if (!message) {
                showError('Please enter a message');
                return;
            }
            
            showLoading('Sending message...');
            
            fetch('/api/demo', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify({ input: message })
            })
            .then(response => response.json())
            .then(data => {
                hideStatus();
                addConversationEntry('user', message);
                addConversationEntry('agent', data.response);
                input.value = '';
                updateMetrics(); // Refresh metrics
            })
            .catch(error => {
                console.error('Error:', error);
                showError('Failed to send message');
            });
        }
        
        function runAutoDemo() {
            showLoading('Running auto demo...');
            
            fetch('/api/demo/auto', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                }
            })
            .then(response => response.json())
            .then(data => {
                hideStatus();
                showSuccess('Auto demo completed successfully!');
                
                // Display results
                const resultsDiv = document.getElementById('auto-demo-results');
                resultsDiv.style.display = 'block';
                resultsDiv.innerHTML = '<h4>Auto Demo Results:</h4>';
                
                data.results.forEach((result, index) => {
                    const resultDiv = document.createElement('div');
                    resultDiv.className = 'auto-demo-result';
                    resultDiv.innerHTML = `
                        <strong>Interaction ${index + 1}:</strong><br>
                        <strong>User:</strong> ${result.input}<br>
                        <strong>Agent:</strong> ${result.response}<br>
                        <strong>Confidence:</strong> ${(result.confidence * 100).toFixed(1)}%
                    `;
                    resultsDiv.appendChild(resultDiv);
                });
                
                updateMetrics(); // Refresh metrics
            })
            .catch(error => {
                console.error('Error:', error);
                showError('Failed to run auto demo');
            });
        }
        
        function addConversationEntry(type, message) {
            const log = document.getElementById('conversation-log');
            const entry = document.createElement('div');
            entry.className = `conversation-entry ${type}`;
            
            const timestamp = new Date().toLocaleTimeString();
            const label = type === 'user' ? 'You' : 'AI Agent';
            
            entry.innerHTML = `
                <strong>${label} (${timestamp}):</strong><br>
                ${message}
            `;
            
            log.appendChild(entry);
            log.scrollTop = log.scrollHeight;
        }
        
        function clearLog() {
            const log = document.getElementById('conversation-log');
            log.innerHTML = '<div class="conversation-entry"><strong>System:</strong> Conversation log cleared.</div>';
            document.getElementById('auto-demo-results').style.display = 'none';
        }
        
        function showLoading(message) {
            const statusDiv = document.getElementById('demo-status');
            statusDiv.innerHTML = `<div class="loading">${message}</div>`;
        }
        
        function showError(message) {
            const statusDiv = document.getElementById('demo-status');
            statusDiv.innerHTML = `<div class="error">${message}</div>`;
        }
        
        function showSuccess(message) {
            const statusDiv = document.getElementById('demo-status');
            statusDiv.innerHTML = `<div class="success">${message}</div>`;
        }
        
        function hideStatus() {
            const statusDiv = document.getElementById('demo-status');
            statusDiv.innerHTML = '';
        }
        
        // Handle Enter key in input
        document.getElementById('user-input').addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                sendMessage();
            }
        });


async function runDataTypeTest() {
    const code = document.getElementById('datatypeInput').value.trim();
    const resultEl = document.getElementById('datatypeResult');
    const statusEl = document.getElementById('datatype-status');

    if (!code) {
        statusEl.innerHTML = '<div class="error">Please enter a Python expression.</div>';
        return;
    }

    statusEl.innerHTML = '<div class="loading">‚è≥ Evaluating...</div>';
    resultEl.textContent = '';

    try {
        const response = await fetch('/api/datatype-test', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ code })
        });

        const data = await response.json();
        statusEl.innerHTML = '';

        if (data.error) {
            resultEl.textContent = `‚ùå ${data.error}`;
            resultEl.style.color = '#f88';
        } else {
            resultEl.style.color = '#0f0';
            resultEl.textContent =
`Input: ${data.input}
Result: ${data.result}
Type: ${data.type}
Literal: ${data.is_literal}`;
        }
    } catch (err) {
        statusEl.innerHTML = '<div class="error">‚ö†Ô∏è Failed to contact server</div>';
    }
}
function runTeachingDemo(){
    showLoading('üìò Teaching: Variables and Data Types...');
    fetch('/api/demo/teach',{method:'POST',headers:{'Content-Type':'application/json'}})
    .then(r=>r.json()).then(d=>{
        hideStatus();
        const div=document.getElementById('teaching-results');
        div.style.display='block';
        div.innerHTML='<h4>üìò Python Data Types & AI Examples</h4>';
        let html='<table><tr><th>Variable</th><th>Value</th><th>Data Type</th><th>AI Use Case</th></tr>';
        d.examples.forEach(e=>{
            let bg='#f9f9f9';
            if(e.type.includes('int')||e.type.includes('float'))bg='#e8f5e8';
            else if(e.type==='str')bg='#e3f2fd';
            else if(e.type==='bool')bg='#fff3e0';
            else if(e.type==='dict'||e.type==='list'||e.type==='tuple')bg='#fce4ec';
            html+=`<tr style="background:${bg};"><td>${e.variable}</td><td>${JSON.stringify(e.value)}</td><td>${e.type}</td><td>${e.ai_use}</td></tr>`;
        });
        html+=`<tr><td colspan="4"><b>${d.summary}</b></td></tr></table>`;
        div.innerHTML+=html;
        showSuccess('‚úÖ Lesson complete! You learned about Python data types.');
    })
    .catch(e=>showError(e));
}

    document.getElementById('datatypeInput').value = '';

        "Type your Python code above and click 'Run Test'";
    document.getElementById('datatype-status').innerHTML = '';
}
