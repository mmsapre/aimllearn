  <script>
    function showStatus(html){ document.getElementById('status').innerHTML = html; }
    function clearResults(){
      document.getElementById('teaching-results').style.display='none';
      document.getElementById('play-output').innerHTML='';
      showStatus('');
    }

    // Return badge color class for type
    function typeBadgeClass(dtype){
      if(['int','float','complex'].includes(dtype)) return 'num';
      if(dtype === 'str') return 'text';
      if(dtype === 'bool') return 'bool';
      if(['list','tuple','set','dict'].includes(dtype)) return 'collection';
      return 'special';
    }

    // --- Teach Data Types ---
    function runTeachingDemo(){
      showStatus('<div class="success">üìò Loading Data Type Lesson...</div>');
      fetch('/api/demo/teach', { method: 'POST', headers: {'Content-Type':'application/json'} })
      .then(r=>r.json())
      .then(d=>{
        const div = document.getElementById('teaching-results');
        div.style.display='block';
        let html = '<h4>üìò Python Data Types Explained</h4>';
        html += '<table><tr><th>Data Type</th><th>Description</th><th>Example</th><th>AI Use Case</th></tr>';
        d.types.forEach(t=>{
          const badgeClass = typeBadgeClass(t.data_type);
          html += `<tr>
            <td><span class="badge ${badgeClass}">${t.data_type}</span></td>
            <td>${t.description}</td>
            <td><code>${t.example}</code></td>
            <td>${t.ai_use}</td>
          </tr>`;
        });
        html += `<tr><td colspan="4"><b>${d.summary}</b></td></tr></table>`;
        html += `<p style="margin-top:10px;font-style:italic;color:#444;">${d.comment}</p>`;
        div.innerHTML = html;
        showStatus('<div class="success">‚úÖ Data Type Lesson Loaded!</div>');
      })
      .catch(e=>showStatus(`<div class="error">${e}</div>`));
    }

    // --- Playground ---
    function runPlayground(){
      const expr = document.getElementById('play-input').value.trim();
      if(!expr){
        document.getElementById('play-output').innerHTML='<div class="error">‚ö† Please enter an expression.</div>';
        return;
      }
      document.getElementById('play-output').innerHTML='<div class="success">‚è≥ Evaluating...</div>';
      fetch('/api/playground',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({expression:expr})})
      .then(r=>r.json())
      .then(d=>{
        if(d.error){
          document.getElementById('play-output').innerHTML=`<div class="error">${d.error}</div>`;
          return;
        }
        const badgeClass = typeBadgeClass(d.data_type);
        let html = `<h4>üß© Playground Result</h4>
        <pre><b>Input:</b> ${d.input}
<b>Evaluated Value:</b> ${d.evaluated_value}
<b>Data Type:</b> <span class="badge ${badgeClass}">${d.data_type}</span>
<b>AI Context:</b> ${d.ai_context}</pre>`;
        document.getElementById('play-output').innerHTML = html;
      })
      .catch(e=>document.getElementById('play-output').innerHTML=`<div class="error">${e}</div>`);
    }
